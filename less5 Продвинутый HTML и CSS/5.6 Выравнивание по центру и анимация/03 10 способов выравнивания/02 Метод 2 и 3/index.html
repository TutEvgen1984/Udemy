Метод 2. top0+bottom0+left0+right0 + margin:auto

        .parent1 {
            position: relative;
        }

        .child1 {
            position: absolute;

            top: 0;
            bottom: 0;
            left: 0;
            right: 0;
            /* после этого мы ничего не видим */

            /* Но если добавим */
            margin: auto;
            /* то элемент выровнялся по центру */
        }

Метод 3. inset: 0; margin: auto;

        .parent2 {
            position: relative;
        }

        .child2 {
            position: absolute;

            inset: 0;
            /* это то же самое что установить top,left,right,bottom = 0 */
            /* https://doka.guide/css/inset/ */

            margin: auto;

        }
